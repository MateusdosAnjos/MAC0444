	SELECT ?Diretor
		WHERE {
		SELECT ?Diretor (count(?Diretor) as ?count)
		WHERE {
			VALUES ?Xp {"Ewan"}.
			VALUES ?Xu {"McGregor"}.
			?x rdfs:subClassOf :Estrela.
			?ator rdf:type ?x.
			?ator :nome_da_estrela ?Xp.
			?ator :nome_familia ?Xu.
			?ator :atua ?atuacao.
			?atuacao :filme_da_atuacao ?Filme.
			?diretor :nome_do_diretor ?PNome.
			?diretor :nome_familia ?PFamilia.
			?Filme :dirigido_por ?diretor.
			BIND(CONCAT(STR(?PNome), " ", STR(?PFamilia))  AS ?Diretor)		
		}
		GROUP BY ?Diretor
		ORDER BY DESC (?count)
		LIMIT 1
		}
