(1)

	SELECT ?TituloDoFilme
	WHERE {
		VALUES ?diretor {:diretor_um}.
		?Filme :dirigido_por ?diretor.
		?Filme :titulo ?TituloDoFilme.
	}
	ORDER BY (?TituloDoFilme)


(2)

	SELECT ?PrimeiroNome ?UltimoNome ?Personagem
	WHERE {
		?filme :titulo "o filme um".
		?ator :estrela_em ?filme.
		?ator :nome_da_estrela ?PrimeiroNome.
		?ator :nome_familia ?UltimoNome.
		?ator :interpreta ?Personagem.
	}
	ORDER BY ?PrimeiroNome ?UltimoNome


(3)

	SELECT ?Filmes ?Diretor ?Ano ?atorUm ?atorDois
	WHERE {
		?atorUm :nome_da_estrela "Artor".
		?atorDois :nome_da_estrela "Artriz".
		?atorUm :estrela_em ?filme.
		?atorDois :estrela_em ?filme.
		?filme :titulo ?Filmes.
		?filme :dirigido_por ?Diretor.
		?filme :ano_lancamento ?Ano
	}
	ORDER BY (?Ano)
